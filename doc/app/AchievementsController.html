def create
  @achievement = Achievement.new(params[:achievement])

  respond_to do |format|
    if @achievement.save

      format.html { redirect_to(achievements_path, :notice => 'Achievement was successfully created.') }
      format.xml  { render :xml => @achievement, :status => :created, :location => @achievement }
    else
      format.html { render :action => "new" }
      format.xml  { render :xml => @achievement.errors, :status => :unprocessable_entity }
    end
  end
end

def update
  @achievement = Achievement.find(params[:id])

  respond_to do |format|
    if @achievement.update_attributes(params[:achievement])
      format.html { redirect_to(achievements_path, :notice => 'Post was successfully updated.') }
      format.xml  { head :ok }
    else
      format.html { render :action => "edit" }
      format.xml  { render :xml => @achievement.errors, :status => :unprocessable_entity }
    end
  end
end
